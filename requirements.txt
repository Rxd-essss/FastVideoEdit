# FastVideoEdit — local YouTube video pipeline
# Python 3.12 recommended. Everything below runs locally; nothing is sent to the cloud.

# --- Transcription (GPU via CUDA, CPU fallback) ---
faster-whisper>=1.2,<2
ctranslate2>=4.0,<5            # inference backend for faster-whisper

# Windows CUDA 12 runtime libraries needed by CTranslate2 (cuBLAS + cuDNN 9).
# No full CUDA Toolkit required; your NVIDIA driver supplies the rest.
# Harmless on a CPU-only machine (wheels just sit unused).
# Pinned to the CUDA-12 majors CTranslate2 4.x is built against (cuBLAS 12.x +
# cuDNN 9.x); an unpinned cuBLAS can pull a version mismatched with cuDNN 9 and
# break with "DLL not found", which then falls back to slow CPU.
nvidia-cublas-cu12>=12.4,<13;  platform_system == "Windows"
nvidia-cudnn-cu12>=9,<10;      platform_system == "Windows"

# --- Config / models / utils ---
pydantic>=2,<3
PyYAML>=6
tqdm>=4

# --- Web review UI (serve.py) ---
fastapi>=0.110,<1
uvicorn[standard]>=0.27,<1

# After install, sanity-check CUDA with:  python scripts/check_gpu_whisper.py
# (confirms faster-whisper loads on the GPU instead of silently using CPU).

# NOTE: the local LLM (bad-takes + chapters) talks to Ollama over plain HTTP

onnxruntime>=1.26,<2
